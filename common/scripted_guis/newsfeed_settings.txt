newsfeed_ensure_settings_initialized_gui = {
    scope = character

    is_valid = {
        is_alive = yes
        NAND = {
            exists = var:newsfeed_migrated_v7
            var:newsfeed_migrated_v7 = yes
        }
    }

    effect = {
        newsfeed_reinitialize_settings = yes

        set_variable = {
            name = newsfeed_migrated_v7
            value = yes
        }
    }
}
