newsfeed_relation_if_not_acquaintance = {
    type = character

    text = {
        trigger = {
            has_RelationToMe_relation = { CHARACTER = scope:second }
        }

        localization_key = "[Localize('CHARACTER_NAME_MY')] [Scope.Char.Custom2('RelationToMe', second.Self)] "
    }

    text = {
        fallback = yes
        localization_key = ""
    }
}
