newsfeed_notify_of_death_effect = {
    if = {
        limit = {
            exists = scope:killer
        }

        send_interface_message = {
            type = newsfeed_death
            title = newsfeed_death_murder
            desc = newsfeed_death_murder_desc
        }
    }
    else = {
        send_interface_message = {
            type = newsfeed_death
            title = newsfeed_death
            desc = newsfeed_death_desc
        }
    }
}
